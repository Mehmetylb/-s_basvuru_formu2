import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      debugShowCheckedModeBanner: false,
      home: JobFormPage(),
    );
  }
}

class JobFormPage extends StatefulWidget {
  const JobFormPage({super.key});

  @override
  State<JobFormPage> createState() => _JobFormPageState();
}

class _JobFormPageState extends State<JobFormPage> {
  final _formKey = GlobalKey<FormState>();

  final nameController = TextEditingController();
  final emailController = TextEditingController();
  final phoneController = TextEditingController();
  final positionController = TextEditingController();
  final experienceController = TextEditingController();

  String result = "";

  void submitForm() {
    if (_formKey.currentState!.validate()) {
      setState(() {
        result =
            "Başvuru Alındı!\n\nAd: ${nameController.text}\nEmail: ${emailController.text}\nTelefon: ${phoneController.text}\nPozisyon: ${positionController.text}\nDeneyim: ${experienceController.text} yıl";
      });
    }
  }

  InputDecoration buildDecoration(String label) {
    return InputDecoration(
      labelText: label,
      border: OutlineInputBorder(
        borderRadius: BorderRadius.circular(10),
      ),
    );
  }

  Widget buildTextField(
      TextEditingController controller, String label, String errorText,
      {TextInputType type = TextInputType.text}) {
    return Padding(
      padding: const EdgeInsets.only(bottom: 12),
      child: TextFormField(
        controller: controller,
        keyboardType: type,
        decoration: buildDecoration(label),
        validator: (value) {
          if (value == null || value.isEmpty) {
            return errorText;
          }
          return null;
        },
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("İş Başvuru Formu"),
        centerTitle: true,
      ),
      body: SingleChildScrollView(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [
            const Text(
              "Lütfen bilgilerinizi eksiksiz doldurun",
              style: TextStyle(fontSize: 16),
            ),
            const SizedBox(height: 20),

            Form(
              key: _formKey,
              child: Column(
                children: [
                  buildTextField(
                      nameController, "Ad Soyad", "Ad soyad giriniz"),

                  buildTextField(emailController, "Email", "Email giriniz",
                      type: TextInputType.emailAddress),

                  buildTextField(phoneController, "Telefon", "Telefon giriniz",
                      type: TextInputType.phone),

                  buildTextField(positionController, "Başvurulan Pozisyon",
                      "Pozisyon giriniz"),

                  buildTextField(experienceController, "Deneyim (yıl)",
                      "Deneyim giriniz",
                      type: TextInputType.number),

                  const SizedBox(height: 20),

                  SizedBox(
                    width: double.infinity,
                    child: ElevatedButton(
                      onPressed: submitForm,
                      child: const Text("Başvur"),
                    ),
                  ),
                ],
              ),
            ),

            const SizedBox(height: 20),

            if (result.isNotEmpty)
              Container(
                padding: const EdgeInsets.all(15),
                decoration: BoxDecoration(
                  color: Colors.green.shade100,
                  borderRadius: BorderRadius.circular(10),
                ),
                child: Text(
                  result,
                  style: const TextStyle(fontSize: 16),
                ),
              ),
          ],
        ),
      ),
    );
  }
}
